Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="amb"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
c25=NETNew["Tinkerforge.BrickletCurrent25",uid,ipcon]
ipcon@Connect[host,port]

(* Get threshold callbacks with a debounce time of 1 second (1000ms) *)
c25@SetDebouncePeriod[1000]

(* Callback for current greater than 5A *)
CurrentReachedCB[sender_,current_]:=
 Print["Current is greater than 5A: "<>ToString[N[Quantity[current,"mA"]]]]
AddEventHandler[c25@CurrentReached,CurrentReachedCB]

(* Configure threshold for "greater than 5A" (unit is mA) *)
option=Tinkerforge`BrickletCurrent25`THRESHOLDUOPTIONUGREATER
c25@SetCurrentCallbackThreshold[option,5*1000,0]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[c25]
ReleaseNETObject[ipcon]
