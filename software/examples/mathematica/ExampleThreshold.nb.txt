Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="XYZ"(*Change to your UID*)

(*Create IPConnection and device object*)
ipcon=NETNew["Tinkerforge.IPConnection"]
c=NETNew["Tinkerforge.BrickletCurrent25",uid,ipcon]
ipcon@Connect[host,port]

(*Get threshold callbacks with a debounce time of 10 seconds (10000ms)*)
c@SetDebouncePeriod[10000]

(*Callback function for current reached callback (parameter has unit mA)*)
CurrentReachedCB[sender_,current_]:=
 Print["Current: "<>ToString[N[Quantity[current,"A/1000"]]]]
AddEventHandler[c@CurrentReached,CurrentReachedCB]

(*Configure threshold for current "greater than 5 A" (unit is mA)*)
option=Tinkerforge`BrickletCurrent25`THRESHOLDUOPTIONUGREATER
c@SetCurrentCallbackThreshold[option,5*1000,0]

Input["Click OK to exit"]

(*Clean up*)
ipcon@Disconnect[]
ReleaseNETObject[c]
ReleaseNETObject[ipcon]
